 14def split(t: np.ndarray) -> Tuple[np.ndarray, int, np.ndarray]:
 15    """
 16    Esta funcion reparte los elementos de t entre dos arrays
 17        con los elementos menores y mayores que el primero.
 18    Args:
 19        t: array con los elementos.
 20    Returns:
 21        Tuple: Una tupla de los elementos menores, el elemento y los mayores.
 22    """
 23
 24    mid = t[0]
 25    # tabla de los menores y tabla de los mayores con list comprehension
 26    t_smallers = [u for u in t if u < mid]
 27    t_greaters = [u for u in t if u > mid]
 28    return t_smallers, mid, t_greaters

 32    """
 33    Esta funcion aplica de manera recursiva Quick Select
 34        usando split.
 35    Args:
 36        t: array con los elementos.
 37        k: posicion del elemento a encontrar (de 0 a len(t).
 38    Returns:
 39        Union: De el valor de k si existe y None si no.
 40    """
 41
 42    # control de errores
 43    if k >= len(t):
 44        return None
 45
 46    if len(t) == 1 and k == 0:
 47        return t[0]
 48
 49    # si la lista es menor que CUTOFF se usa np.sort
 50    if len(t) <= CUTOFF:
 51        return np.sort(t)[k]
 52
 53    t_smallers, mid, t_greaters = split(t)
 54    m = len(t_smallers)
 55

 58    elif k < m:

128def pivot5(t: np.ndarray) -> int:
129    """
130    Esta funcion devuelve el pivote.
131    Args:
132        t: array con los elementos.
133    Returns:
134        pivot: el pivote.
135    """
136
137    # division para comprobar en cuantos grupos se parte
138    # y si hay algun grupo que no alcanza el groupsize
139    n, remain = divmod(len(t), GROUPSIZE)
140
141    # posicion mediana del array
142    k = int(ceil(GROUPSIZE/2)) - 1
143
144    medians = [np.sort(t[j * GROUPSIZE: (j+1) * GROUPSIZE])[k]
145               for j in range(0, n)]
146
147    if remain:
148        k = int(ceil(remain/2)) - 1
149        medians.append(np.sort(t[-remain:])[k])
150
151    # posicion mediana de las medianas
152    k = int(ceil(len(medians)/2)) - 1
153
154    # si la lista es menor que CUTOFF se usa np.sort
155    if len(medians) <= CUTOFF:
156        pivot = np.sort(medians)[k]
157    else:
158        pivot = qsel5_nr(medians, k)
159
160    return pivot

163def qsel5_nr(t: np.ndarray, k: int) -> Union[int, None]:
164    """
165    Esta funcion aplica de manera no recursiva Quick Select 5
166        usando split_pivot y pivot5.
167    Args:
168        t: array con los elementos.
169        k: posicion del elemento a encontrar (de 0 a len(t)).
170    Returns:
171        Union: De el valor de k si existe y None si no.
172    """
173
174    # control de errores
175    if k >= len(t) or k < 0:
176        return None
177
178    aux_t = t.copy()
179    while len(aux_t) > 1 or k > 0:
180
181        # si la lista es menor que CUTOFF se usa np.sort
182        if len(aux_t) <= CUTOFF:
183            return np.sort(aux_t)[k]
184
185        # busqueda del pivote y particion de la lista
186        mid = pivot5(aux_t)
187        t_smallers, pivot_value, t_greaters = split_pivot(aux_t, mid)
188
189        # misma logica que el quick select tradicional
190        m = len(t_smallers)
191        if k == m:
192            return pivot_value
193
194        elif k < m:
195            aux_t = t_smallers
196        else:
197            k = k - m-1
198            aux_t = t_greaters
199
200    return aux_t[0]

232def change_pd(c: int, l_coins: list[int]) -> np.ndarray:
233    """
234    Esta funcion aplica el algoritmo change con
235        programacion dinamica.
236    Args:
237        c: cantidad optima de monedas.
238        l_coins: lista de monedas.
239    Returns:
240        ndarray: La array generada.
241    """
242    # En n obtenemos la cantidad de monedas en la lista
243    n = len(l_coins)
244
245    # Creamos una matriz llena de ceros de tam cantidad de monedas / cantidad optima
246    dp = np.zeros((n + 1, c + 1), dtype=int)
247
248    # Inicializamos la primera fila de la matriz con valores cantidad optima + 1
249    for i in range(1, c + 1):
250        dp[0][i] = c + 1
251
252    for i in range(1, n + 1):
253        for j in range(1, c + 1):
254            if l_coins[i - 1] <= j:
255                # Calculamos el minimo entre los dos valores de la matriz si es menor:
256                dp[i][j] = min(1 + dp[i][j - l_coins[i - 1]], dp[i - 1][j])
257            else:
258                # Copiamos el valor de la fila anterior si es mayor
259                dp[i][j] = dp[i - 1][j]
260
261    return dp

264def optimal_change_pd(c: int, l_coins: List[int]) -> Dict[int, int]:
265    """
266    Calcula la cantidad optima de monedas para el cambio.
267    Args:
268        c: cantidad optima de monedas.
269        l_coins: lista de monedas.
270    Returns:
271        ndarray: Diccionario que contiene las monedas utilizadas y sus cantidades
272    """
273
274    # Obtenemos la cantidad de monedas en la lista
275    n = len(l_coins)
276
277    # Usamos change_pd para generar la matriz
278    dp = change_pd(c, l_coins)
279
280    # Creamos un diccionario ordenado para almacenar las cantidades de cada tipo de moneda
281    coin_counts = OrderedDict()
282
283    i, j = n, c
284    while i > 0 and j > 0:
285        # Si los valores de la matriz son iguales, pasamos a la siguiente fila en la matriz
286        if dp[i][j] == dp[i - 1][j]:
287            i -= 1
288        else:
289            if l_coins[i - 1] <= j:
290                # Si la moneda actual es menor o igual al tam actual de la lista,
291                # agregamos la moneda al conteo y actualizamos
292                if l_coins[i - 1] in coin_counts:
293                    coin_counts[l_coins[i - 1]] += 1
294                else:
295                    coin_counts[l_coins[i - 1]] = 1
296                j -= l_coins[i - 1]
297    return coin_counts

300def knapsack_fract_greedy(l_weights: List[int], l_values: List[int], bound: int) -> Dict[int, float]:
301    """
302    Esta funcion resuelve el problema de la mochila fraccionaria utilizando un enfoque codicioso
303    Calcula los pesos a tomar de cada elemento en la solucion greedy del problema de la mochila fraccionaria.
304    Args:
305        l_weights: Lista de pesos de los elementos
306        l_values: Lista de valores de los elementos
307        bound: limite del peso de la mochila
308    Returns:
309        ndarray: La array generada.
310    """
311    # Diccionario para almacenar los pesos a tomar de cada elemento
312    diccionario = dict()
313
314    # Lista de tuplas con peso, valor e √≠ndice
315    lista = list(zip(l_weights, l_values, range(len(l_weights))))
316
317    # Ordenamos la lista por la relaci√≥n valor-peso (mayor a menor)
318    lista.sort(key=lambda x: x[1] / x[0], reverse=True)
319
320    for i in range(len(lista)):
321        id_obj = lista[i][2]
322        if bound <= 0:
323            # Si el l√≠mite de peso es menor o igual a cero, asigna un peso de 0 al elemento actual
324            diccionario[id_obj] = 0
325
326        if lista[i][0] <= bound:
327            # Si el peso del elemento cabe en el l√≠mite, se asigna su peso completo
328            diccionario[id_obj] = float(lista[i][0])
329            # Restamos el peso del elemento del l√≠mite total disponible
330            bound -= lista[i][0]
331        else:
332            # Calculamos la fracci√≥n del peso que cabe en el l√≠mite
333            diccionario[id_obj] = float(bound / lista[i][0] * lista[i][0])
334            bound = 0
335
336    # Ordena el diccionario por las claves
337    diccionario = dict(OrderedDict(sorted(diccionario.items())))
338
339    return diccionario

342def knapsack_01_pd(l_weights: List[int], l_values: List[int], bound: int) -> int:
343    """
344    Resuelve el problema de la mochila 0-1 y calcula el valor optimo de la mochila 0-1 obtenido mediante programacion dinamica.
345    Args:
346        l_weights: Lista de pesos de los elementos.
347        l_values: Lista de valores de los elementos.
348        bound: limite de peso de la mochila.
349    Returns:
350        int: El valor optimo de la mochila
351    """
352
353    # Inicializamos la matriz
354    dp = np.zeros((len(l_weights), bound + 1), dtype=int)
355
356    for i in range(len(l_weights)):
357        for j in range(bound + 1):
358
359            if j == 0:
360                # Si el peso de la mochila es 0, el valor es 0
361                dp[i][j] = 0
362
363            elif i == 0:
364                # Si el peso del elemento es menor o igual al peso de la mochila
365                if l_weights[i] <= j:
366                    # Asignamos el valor del elemento
367                    dp[i][j] = l_values[i]
368
369            # Si el peso del elemento actual es mayor que el peso disponible de la mochila
370            elif l_weights[i] > j:
371                # Copiamos el valor de la celda de arriba
372                dp[i][j] = dp[i - 1][j]
373
374            else:
375                # Tomamos el m√°ximo entre el valor sin el elemento actual y con el elemento actual
376                dp[i][j] = max(
377                    dp[i - 1][j],
378                    l_values[i] + dp[i - 1][j - l_weights[i]])
379
380    return dp[len(l_weights) - 1][bound]
